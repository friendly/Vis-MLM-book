% Pandoc's \citeproc{key}{text} uses \begingroup\cite{key}\endgroup.
% On the first LaTeX pass (or when .aux is stale from a failed build),
% undefined citations inside \footnote{} cause "Extra }, or forgotten \endgroup".
% \renewcommand does not reliably override \NewDocumentCommand-defined commands
% in modern LaTeX (xparse internal dispatch). Use \def at the TeX primitive level.
% Pandoc provides the correctly formatted text in #2 — \cite{#1} is redundant
% for Pandoc citeproc documents (no BibTeX step; bibliography is already in .tex).
\def\citeproc#1#2{#2}

% R functions — two args: {sort-key}{display-name}
% Caller includes () in both args; \ixfunc does NOT add them.
% Sort key: plain name with raw _ (OK for makeindex sorting; not typeset)
% Display:   name with \_ escaping so LaTeX can typeset it from the .ind file
% e.g. \ixfunc{stat_ellipse()}{stat\_ellipse()}
% Single-line definitions: parse-latex (pandoc filter) expands these macros during
% the PDF pandoc pass and inserts the body literally. Multi-line bodies with %
% at line-ends produce %.\footnote{} when a footnote follows — commenting out the
% footnote opener. Single-line bodies expand safely with no embedded %.
\newcommand{\ixfunc}[2]{\index{#1@\texttt{#2}}}

% R packages:  indexed under both package name and packages!
\newcommand{\ixp}[1]{\index{#1@\texttt{#1} package}\index{packages!#1@\texttt{#1}}}

% data sets: \ixd{name} — indexes under name and datasets!name, both with \texttt
\newcommand{\ixd}[1]{\index{#1@\texttt{#1} data}\index{datasets!#1@\texttt{#1}}}
