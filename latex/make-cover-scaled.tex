% Wrapper to produce a letter-sized cover PDF from the A4 source image.
% The source (cover-peng.pdf) is A4 (8.27" x 11.69"); the book is letterpaper
% (8.5" x 11"). Scaling to fit by height leaves ~0.36" equal side margins.
% Compile from project root with:
%   xelatex -output-directory=images/cover latex/make-cover-scaled.tex
\documentclass{article}
\usepackage[
  paperwidth=8.5in,
  paperheight=11in,
  margin=0pt,
  noheadfoot
]{geometry}
\usepackage{graphicx}
\pagestyle{empty}
\begin{document}
\noindent\makebox[\paperwidth][c]{%
  \includegraphics[height=\paperheight,keepaspectratio]%
    {images/cover/cover-peng.pdf}%
}%
\end{document}
