% Source for images/fig-book-parts.{pdf,png} — the four-part book overview
% diagram shown in index.qmd (chunk `fig-book-parts`). This is the single
% source of truth: the smartdiagram code originally lived in a disabled
% {tikz} chunk in index.qmd and was frozen as a low-res PNG (1307x145 px,
% ~251 effective dpi); see issues/CMYK-checklist.md.
%
% Not built by make-diagrams.sh (that script handles the part-page and
% chapter-opener mindmaps only). Rebuild both outputs from this directory:
%
%   # vector PDF, CMYK colors — used in the print/PDF book
%   xelatex -jobname fig-book-parts "\def\cmykcolors{}\input{fig-book-parts.tex}"
%   mv fig-book-parts.pdf ../../images/
%
%   # high-res sRGB PNG (600 dpi) — used in the HTML book
%   xelatex -jobname fig-book-parts-rgb "\input{fig-book-parts.tex}"
%   magick -density 600 fig-book-parts-rgb.pdf ../../images/fig-book-parts.png
%   rm -f fig-book-parts-rgb.pdf *.aux *.log
%
% \cmykcolors makes xcolor mix all colors in the cmyk model, matching the
% CMYK requirement for CRC print production (see issues/CMYK-colors.md).
\ifdefined\cmykcolors
  \PassOptionsToPackage{cmyk}{xcolor}
\else
  \PassOptionsToPackage{rgb}{xcolor}
\fi
\documentclass[border=2pt]{standalone}
\usepackage{smartdiagram}

\begin{document}
\smartdiagramset{
  back arrow disabled=true,
  module minimum width=2cm,
  module minimum height=1.2cm,
  text width=1.8cm,
  module x sep=3,
  font=\small,
  set color list={blue!35, teal!40, green!35, orange!40},
}
\smartdiagram[flow diagram:horizontal]{
  Orienting\\ Ideas,
  Exploratory\\ Methods,
  Univariate\\ Models,
  Multivariate\\ Models
}
\end{document}
