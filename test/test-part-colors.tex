% Switch partcol at each \part boundary.
% Patch \@part (not \part) so the color change fires AFTER \cleardoublepage
% has flushed pending pages — otherwise the last pages of the previous part
% are output with the new part's color already set.
\newcounter{partnum}
\makeatletter
\pretocmd{\@part}{%
  \stepcounter{partnum}%
  \ifcase\value{partnum}
    \or \colorlet{partcol}{partI}%
    \or \colorlet{partcol}{partII}%
    \or \colorlet{partcol}{partIII}%
    \or \colorlet{partcol}{partIV}%
    \or \colorlet{partcol}{black}%
  \fi
  \colorlet{partcol80}{partcol!80}%
  \colorlet{partcol50}{partcol!50}%
  \colorlet{partcol20}{partcol!20}%
}{}{}
\makeatother

% Mirror krantz's PageNumFont, RunningHeadFont, and even/odd head layout.

% are colored even though \@the@page fires before \RunningHeadFont.
\makeatletter
\newcommand\PageNumFont{\reset@font\color{partcol}\fontsize{10}{12}\selectfont}

\def\ps@colorheadings{%
  \let\@oddfoot\@empty\let\@evenfoot\@empty
  \def\@evenhead{{\@the@page\RunningHeadFont\hfill\leftmark}}%
  \def\@oddhead{\RunningHeadFont\rightmark\hfill\PageNumFont\thepage}%
}
